124 static inline bool IsImplRessonance( int rss_id ) {
125
126 bool is_p33 = IsP33(rss_id);
127 bool is_f37 = IsF37(rss_id);
128 bool is_nstar = IsP11(rss_id) || IsD13(rss_id) || IsF15(rss_id) || IsS11(rss_id);
129 return ( is_p33 || is_f37 || is_nstar );
130 }
131 static inline bool IsLepton(int pdgId){
132 const Int_t electron = 11;
133 const Int_t mu = 13;
134 const Int_t mu_p = -13;
135 const Int_t nu_mu = 14;
136 const Int_t nu_mu_bar = -14;
137 return ( pdgId == electron || pdgId == mu || pdgId == mu_p || pdgId == nu_mu || pdgId == nu_mu_bar );
138 }
