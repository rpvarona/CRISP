164 friend ostream& operator<<(ostream& out, TStoragef& stor){
165
166 out.setf(ios::scientific, ios::floatfield);
167
168 out << "\n Physics Properties [Breakup System] \n\n";

170 out <<" <Temperature> : " << stor.Temperature <<" "

172 out <<" <Entropy> : " << stor.Entropy <<" "
173 << stor.GetsigEntropy()<< "\n";
174 out <<" <Multiplicity> : " << stor.Multiplicity<<" "
175 << stor.GetsigMultiplicity()<< "\n";
176 out <<" <Cv> : " << stor.Cv <<" "
177 << stor.GetsigCv()<< "\n";
178
179 out << "\n\n";
180
181 out << " Prodction [Nuclide(Nucleon)] \n\n";
182 out << " ZAID Yield RERR \n\n";
183
184 out.adjustfield;
185
186 map<int, double>::iterator imap;
187
188 for (imap = stor.Yield.begin(); imap != stor.Yield.end(); imap++ ){
189 out << " "<< imap->first
190 << " "<< imap->second
191 << " "<< sqrt(- imap->second * imap->second + stor.Yield2[imap->first])
192 << "\n";
193 }
194
195 return out;
196 }
